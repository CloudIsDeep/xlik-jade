### 监听器

```lua
--- 恢复生命监听器
---@param obj Unit
monitor.create(
    "hp_regen",
    0.2,
    function(obj)
        local regen = 0.2 * obj:hpRegen()
        if (regen >= 0) then
            obj:hpCur("+=" .. regen)
        else
            obj:hpCur("-=" .. regen)
        end
    end,
    function(obj)
        return obj:isDead() or nil == obj:hpRegen() or 0 == obj:hpRegen()
    end)
```
